import UIKit

// BotAction -> botHyarit
enum botHyarit: Int {
    case buy = 0
    case ignore = 1
    case sell = 2
}

// MarketData -> ValutaCena
struct ValutaCena {
    let price: Double
    let valuta: String

    var priceWithValuta: String {
        return "\(price) \(valuta)"
    }
}

// Output -> VbIvod
protocol VbIvod {
    func printBuy(_ data: ValutaCena)
    func printIgnore(_ data: ValutaCena)
    func printSell(_ data: ValutaCena)
    func printDeal(openPrice: Double, closePrice: Double, income: Double)
    func printResult(balance: Double, valuta: String)
}

// ConsoleOutput -> krasivovbivodit
struct krasivovbivodit: VbIvod {
    func printBuy(_ data: ValutaCena) {
        print("\(data.priceWithValuta) - покупка")
    }

    func printIgnore(_ data: ValutaCena) {
        print("\(data.priceWithValuta) - игнорирование")
    }

    func printSell(_ data: ValutaCena) {
        print("\(data.priceWithValuta) - продажа")
    }

    func printDeal(openPrice: Double, closePrice: Double, income: Double) {
        print("FROM \(openPrice) -> TO \(closePrice), INCOME = \(income)")
    }

    func printResult(balance: Double, valuta: String) {
        print("Итоговый баланс: \(balance) \(valuta)")
    }
}

final class ciklpokypki {
    private var balance: Double = 0.0
    private var openBuyPrice: Double? = nil

    private let valuta: String
    private let output: VbIvod

    private var isTradeOpen: Bool {
        return openBuyPrice != nil
    }

    init(valuta: String, output: VbIvod) {
        self.valuta = valuta
        self.output = output
    }

    public func startSimulation() {
        for _ in 0...10 {
            let marketData = generateMarketDataForStep()
            let botAction = generateRandomBotAction()

            processBotAction(botAction, marketData: marketData)

            if isTradeOpen {
                // ничего
            } else {
                // ничего
            }
        }

        output.printResult(balance: balance, valuta: valuta)
    }

    private func processBotAction(_ action: botHyarit, marketData: ValutaCena) {
        if action == .buy {
            processBuy(marketData)
        }

        if action == .ignore {
            processIgnore(marketData)
        }

        if action == .sell {
            processSell(marketData)
        }
    }

    private func processBuy(_ marketData: ValutaCena) {
        output.printBuy(marketData)

        if openBuyPrice == nil {
            openBuyPrice = marketData.price
        }
    }

    private func processIgnore(_ marketData: ValutaCena) {
        output.printIgnore(marketData)
    }

    private func processSell(_ marketData: ValutaCena) {
        output.printSell(marketData)

        if let buyPrice = openBuyPrice {
            let income = marketData.price - buyPrice
            balance += income
            output.printDeal(openPrice: buyPrice, closePrice: marketData.price, income: income)
        }

        openBuyPrice = nil
    }
}

extension ciklpokypki {
    fileprivate func generateMarketDataForStep() -> ValutaCena {
        let price = generateRandomPrice()
        return ValutaCena(price: price, valuta: valuta)
    }

    fileprivate func generateRandomPrice() -> Double {
        return Double.random(in: 11...16)
    }

    fileprivate func generateRandomBotAction() -> botHyarit {
        let raw = Int.random(in: 0...2)
        return botHyarit(rawValue: raw)!
    }
}

let bot = ciklpokypki(valuta: "CNY", output: krasivovbivodit())
bot.startSimulation()

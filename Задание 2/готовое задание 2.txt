// Trading Bot Simulation

import Foundation

// Protocol defining a tradeable asset
protocol Tradeable {
    var symbol: String { get }
    var price: Double { get set }
}

// Struct representing a Stock
struct Stock: Tradeable {
    var symbol: String
    var price: Double
}

// Enum representing the type of order
enum OrderType {
    case market
    case limit
}

// Class representing a Trading Order
class Order {
    var asset: Tradeable
    var quantity: Int
    var orderType: OrderType
    var status: String = "Pending"
    
    init(asset: Tradeable, quantity: Int, orderType: OrderType) {
        self.asset = asset
        self.quantity = quantity
        self.orderType = orderType
    }
    
    func execute() {
        // Simulate the order execution
        status = "Executed"
        print("Executed order for \(quantity) of \(asset.symbol) at price \(asset.price)")
    }
}

// Class representing a Trading Bot
class TradingBot {
    var name: String
    var balance: Double
    var portfolio: [Tradeable] = []
    
    init(name: String, balance: Double) {
        self.name = name
        self.balance = balance
    }
    
    func buy(asset: Tradeable, quantity: Int) {
        let order = Order(asset: asset, quantity: quantity, orderType: .market)
        if balance >= asset.price * Double(quantity) {
            order.execute()
            portfolio.append(asset)
            balance -= asset.price * Double(quantity)
        } else {
            print("Insufficient funds to buy \(quantity) of \(asset.symbol)")
        }
    }
}

// Testing the Trading Bot
let appleStock = Stock(symbol: "AAPL", price: 150.0)
let bot = TradingBot(name: "MyTradingBot", balance: 1000.0)
bot.buy(asset: appleStock, quantity: 5)  // Should execute
